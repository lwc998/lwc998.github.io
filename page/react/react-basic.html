组件

// 类式组件
import React, { Component } from 'react';

export default class ClassComponent extends Component {
  render() {
    return (
      <div>ClassComponent</div>
    )
  }
}

// 函数式组件
import React from 'react';

export default function FunctionComponent() {
  return (
    <div>FunctionComponent</div>
  )
}
  • 从代码的角度上看
    • Class 组件渲染的是 render 函数中的 return 的内容；
    • Function 组件渲染的是，函数本身返回的内容

State 和 Props

// 类组件
import React, { Component } from 'react';

export default class ClassComponent extends Component {
  constructor(props) {
    super(props);

    this.state = {
      num: 0,
      msg: 'hello'
    }
  }

  changNum = function(type) {
    this.setState({
      num: type === 'increase' ? this.state.num + 1 : this.state.num - 1
    }, () => {
      // cb 中可以直接拿到更新后的值
      console.log(this.state.num, 'cb')
    })
    console.log(this.state.num, 'before cb')
  }

  changNumArrow = (type) => {
    this.setState({
      num: type === 'increase' ? this.state.num + 1 : this.state.num - 1
    }, () => {
      // cb 中可以直接拿到更新后的值
      console.log(this.state.num, 'cb')
    })
    console.log(this.state.num, 'before cb')
  }

  handleChange = function(e) {
    this.setState({
      msg: e.target.value
    })
  }

  render() {
    const {num} = this.state;
    const {name} = this.props;
    return (
      <h3>类组件</h3>
      <div>{num} {name}</div>
      // 改变状态
      <button onClick={this.changNum.bind(this,'increase')}>+ 1</button>
      <button onClick={() => this.changNum('increase')}>+ 1</button>
      <button onClick={this.changNumArrow('increase')}>+ 1</button>

      // 受控组件
      <input type="text" value={msg} onChange={this.handleChange} />
    )
  }
}

// 函数式组件
import React, { useState } from "react";

export default function FunComponent({ propsData }) {
  const [num, setNum] = useState(0);
  const [msg, setMsg] = useState("hello");

  const changNum = (type) => {
    setNum(type === "increase" ? num + 1 : num - 1);
  };
  const changeMsg = (e) => {
    setMsg(e.target.value);
  };

  return (
    <div>
      <h3>函数式组件 {propsData}</h3>
      <div>number: {num}</div>
      <button onClick={() => changNum("increase")}>+1</button>
      <button onClick={() => changNum("decrease")}>+1</button>
      {/* 受控组件 */}
      <input type="text" value={msg} onChange={changMsg} />
    </div>
  );
}

hooks

  • hooks 是 react16.8 版本引入的新特性，用来解决类组件中状态和生命周期的问题。
  1. useState 用来创建状态
import React, { useState } from "react";

export default function Demo() {
  const [num, setNum] = useState(0);
  const addNum = () => {
    setNum(num + 1);
    setNum((initData) => {
      console.log(initData);
      return initData + 1;
    });
  };
  return (
    <>
      <div>{num}</div>
      <button onClick={addNum}>+1</button>
    </>
  );
}
  1. useEffect 用来创建生命周期
import React, { useEffect, useState } from "react";

export default function Demo(props) {
  const [state, setState] = useState(() => {
    console.log("getDerivedStateFromProps");
    return "";
  });

  useEffect(() => {
    console.log("componentDidMount");
    // 组件挂载时执行: 可做数据请求
    new Promise((resolve) => {
      setTimeout(() => {
        resolve("success");
      }, 300);
    }).then((res) => {
      setState(res);
    });

    return () => {
      // 组件卸载时执行: 可做一些监听事件接触等
      console.log("componentWillUnmount");
    };
  }, []);

  useEffect(() => {
    console.log("componentDidUpdate");
  });

  useEffect(() => {
    console.log("componentWillReceiveProps");
  }, [props]);

  return (
    <div>
      <h2>模拟生命周期：{state}</h2>
      <button onClick={() => setState((v) => `${Math.random()}`)}>
        update
      </button>
    </div>
  );
}

import React, { useLayoutEffect, useState } from "react";

export default function Demo() {
  const ref = useRef(null);
  const [tooltipHeight, setTooltipHeight] = useState(0);

  useLayoutEffect(() => {
    const { height } = ref.current.getBoundingClientRect();
    setTooltipHeight(height);
  }, []);
  return <div ref={ref}>{tooltipHeight}</div>;
}

  • 获取自定义组件的 ref
import React, { useRef, forwardRef, useImperativeHandle } from "react";

export default function Demo() {

  const inputRef = useRef(null);

  const handleFocus = () => {
    inputRef.current.focus();
  }
  return (
    <>
      <inputModel ref={inputRef} />
      <inputModel2 ref={inputRef} />
      <button onClick={handleFocus}></button>
    </>
  )
}

// forwardRef 允许组件使用 ref 将DOM节点暴露给父组件，可多组件套用
const inputModel = forwardRef(ChildInput);
const inputModel2 = forwardRef(ChildInput2);

// 1. 父组件直接操作DOM
function ChildInput(props, ref) =>{
  const { label, otherProps } = props;
  return (
    <label>
      {label}
      // <input value={value} onChange={onChange} ref={ref} />
      <input {...otherProps} ref={ref} />
    </label>
  )
}

// 2. 父组件操作子组件的方法，通过 useImperativeHandle 传递API
function ChildInput2(props, ref) =>{
  const inputRef = useRef(null);
  useImperativeHandle(ref, () => ({
    return {
      focus() {
        inputRef.current.focus();
      }
    }
  }))
  return (
    <input ref={inputRef} />
  )
}
